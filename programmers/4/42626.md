```
#include <string>
#include <vector>
#include <queue>

using namespace std;

priority_queue<int,vector<int>,greater<int>> q;

int solution(vector<int> scoville, int K) {
    int answer = 0;
    
    for(int i = 0; i < scoville.size(); i++){
        q.push(scoville[i]);
    }
    
    while(q.top() < K){
        int a = q.top();
        q.pop();
        
        if(q.empty()){
            return -1;
        }
        
        int b = q.top();
        q.pop();
        
        a = a + b*2;
        
        q.push(a);
        
        answer++;
    }
    return answer;
}
```
