int dfs(const vector<int> &num, int target, int current_sum, int index) {
    if (index == num.size()) {
        return (current_sum == target) ? 1 : 0;
    }

    return dfs(num, target, current_sum + num[index], index + 1) +
           dfs(num, target, current_sum - num[index], index + 1);
}
